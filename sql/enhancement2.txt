/* 
 This changes the words Nylon Rope to Climbing Rope
 */

UPDATE inventory
SET invDescription=REPLACE(invDescription, 'nylon rope', 'climbing rope')
WHERE 1;
UPDATE inventory
SET invName=REPLACE(invName, 'Nylon Rope', 'Climbing Rope')
WHERE 1
